using System;
using UnityEngine;

public class MoveScript
{
    //This script move monobehaviour object

    public float speed = 5;
    public float acceleration = 1;
    
    public void Awake()
    {
        if(speed == 0 || acceleration == 0)
            return;
    }
        
    public void Start()
    {

    }

    public void Update()
    {
        transform.position == transform.forward * speed * Time.deltaTime;
    }
}
